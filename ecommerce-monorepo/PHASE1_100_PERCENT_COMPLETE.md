**OrderListScreen & OrderDetailScreen:**
```typescript
// Add to apiClient (mobile/src/api/client.ts):
async getOrders() {
  const response = await client.get('/orders')
  return response.data
}

async getOrder(orderId: string) {
  const response = await client.get(`/orders/${orderId}`)
  return response.data
}
```

**ProductListScreen:**
```typescript
// Add to apiClient:
async getProducts(page = 1, limit = 10, category?: string, search?: string) {
  const params = new URLSearchParams()
  params.append('page', page.toString())
  params.append('limit', limit.toString())
  if (category) params.append('category', category)
  if (search) params.append('search', search)
  const response = await client.get(`/products?${params}`)
  return response.data
}

async getProduct(productId: string) {
  const response = await client.get(`/products/${productId}`)
  return response.data
}
```

**SearchScreen:**
```typescript
// Add to apiClient:
async search(query: string, type?: 'product' | 'service') {
  const params = new URLSearchParams()
  params.append('q', query)
  if (type) params.append('type', type)
  const response = await client.get(`/search?${params}`)
  return response.data
}
```

**NotificationsScreen:**
```typescript
// Add to apiClient:
async getNotifications() {
  const response = await client.get('/notifications')
  return response.data
}

async markNotificationRead(notificationId: string) {
  const response = await client.put(`/notifications/${notificationId}/read`)
  return response.data
}

async markAllNotificationsRead() {
  const response = await client.put('/notifications/read-all')
  return response.data
}
```
